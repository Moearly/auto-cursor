```typescript
interface UsageContextType {
  authInfo: UserAuthInfo | null;
  usageData: AggregatedUsageData | null;
  loading: boolean;
  error: string | null;
  refreshAuth: () => Promise<void>;
  refreshUsage: () => Promise<void>;
}

const UsageContext = createContext<UsageContextType | undefined>(undefined);

export const UsageProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [authInfo, setAuthInfo] = useState<UserAuthInfo | null>(null);
  const [usageData, setUsageData] = useState<AggregatedUsageData | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const refreshAuth = async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await UsageService.checkAuth();
      setAuthInfo(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'æœªçŸ¥é”™è¯¯');
    } finally {
      setLoading(false);
    }
  };

  const refreshUsage = async () => {
    setLoading(true);
    setError(null);
    try {
      const result = await UsageService.getAggregatedUsage();
      setUsageData(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'æœªçŸ¥é”™è¯¯');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    refreshAuth();
  }, []);

  return (
    <UsageContext.Provider
      value={{
        authInfo,
        usageData,
        loading,
        error,
        refreshAuth,
        refreshUsage
      }}
    >
      {children}
    </UsageContext.Provider>
  );
};

export const useUsage = () => {
  const context = useContext(UsageContext);
  if (context === undefined) {
    throw new Error('useUsage must be used within a UsageProvider');
  }
  return context;
};
```

```typescript
// é˜²æŠ–
export function useDebounce<T extends (...args: any[]) => any>(
  callback: T,
  delay: number
) {
  const timeoutRef = useRef<NodeJS.Timeout>();

  return useCallback((...args: Parameters<T>) => {
    if (timeoutRef.current) {
      clearTimeout(timeoutRef.current);
    }
    timeoutRef.current = setTimeout(() => {
      callback(...args);
    }, delay);
  }, [callback, delay]);
```
